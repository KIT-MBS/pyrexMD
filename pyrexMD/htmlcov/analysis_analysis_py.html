109def alignto(mobile, ref, sel1, sel2, weights='mass', tol_mass=0.1, strict=False): 

110 """ 

111 Modified version of MDAnalysis.analysis.align.alignto() which works properly 

112 with two different selection strings. 

113 

114 Description: 

115 Perform a spatial superposition by minimizing the RMSD. 

116 

117 Spatially align the group of atoms `mobile` to `reference` by doing a 

118 RMSD fit on `sel1` of mobile atoms and `sel1` of referece atoms. 

119 

120 The superposition is done in the following way: 

121 

122 1. A rotation matrix is computed that minimizes the RMSD between 

123 the coordinates of `mobile.select_atoms(sel1)` and 

124 `reference.select_atoms(sel2)`; before the rotation, `mobile` is 

125 translated so that its center of geometry (or center of mass) 

126 coincides with the one of `reference`. (See below for explanation of 

127 how *sel1* and *sel2* are derived from `select`.) 

128 2. All atoms in :class:`~MDAnalysis.core.universe.Universe` that 

129 contain `mobile` are shifted and rotated. (See below for how 

130 to change this behavior through the `subselection` keyword.) 

131 

132 Args: 

133 mobile (universe): mobile structure 

134 ref (universe): reference structure 

135 sel1 (str): selection string of mobile structure 

136 sel2 (str): selection string of reference structure 

137 weights (None, str, array): 

138 | None: weigh each atom equally 

139 | "mass": weigh atoms based on mass 

140 | array: If a float array of the same length as `mobile` is provided, 

141 use each element of the `array_like` as a weight for the corresponding 

142 atom in `mobile`. 

143 tol_mass (float): Reject match if the atomic masses for matched atoms 

144 differ by more than `tol_mass`, default [0.1] 

145 strict (bool): 

146 | True: Will raise :exc:`SelectionError` if a single atom does not 

147 match between the two selections. 

148 | False: Will try to prepare a matching selection by dropping residues 

149 with non-matching atoms. 

150 

151 Returns: 

152 old_rmsd (float) 

153 RMSD before spatial alignment 

154 new_rmsd (float) 

155 RMSD after spatial alignment 

156 """ 

157 mobile_atoms = mobile.select_atoms(sel1) 

158 ref_atoms = ref.select_atoms(sel2) 

159 mobile_atoms, ref_atoms = _align.get_matching_atoms(mobile_atoms, ref_atoms, 

160 tol_mass=tol_mass, 

161 strict=strict) 

162 

163 weights = _align.get_weights(ref_atoms, weights) 

164 mobile_com = mobile_atoms.center(weights) 

165 ref_com = ref_atoms.center(weights) 

166 

167 mobile_coordinates = mobile_atoms.positions - mobile_com 

168 ref_coordinates = ref_atoms.positions - ref_com 

169 

170 old_rmsd = _rms.rmsd(mobile_coordinates, ref_coordinates, weights) 

171 # _fit_to DOES subtract center of mass, will provide proper min_rmsd 

172 mobile_atoms, new_rmsd = _align._fit_to(mobile_coordinates, ref_coordinates, 

173 mobile_atoms, mobile_com, ref_com, 

174 weights=weights) 

175 return old_rmsd, new_rmsd

178def get_rmsd(mobile, ref, sel1, sel2, prec=3, weights='mass', superposition=True): 

179 """ 

180 Returns rmsd for single frame. 

181 

182 Args: 

183 mobile (universe, atomgrp): mobile structure 

184 ref (universe, atomgrp): reference structure 

185 sel1 (str): selection string of mobile structure 

186 sel2 (str): selection string of reference structure 

187 prec (None, int): rounding precission 

188 weights (bool, str, array_like): 

189 | weights during superposition of mobile and reference 

190 | None: no weights 

191 | "mass": atom masses as weights 

192 | array_like: If a float array of the same length as "mobile" is 

193 provided, use each element of the "array_like" as a weight for the 

194 corresponding atom in "mobile". 

195 superposition (bool) 

196 

197 Returns: 

198 rmsd (float) 

199 rmsd value in Å 

200 """ 

201 mobile_atoms = mobile.select_atoms(sel1) 

202 ref_atoms = ref.select_atoms(sel2) 

203 rmsd = _rms.rmsd(mobile_atoms.positions, ref_atoms.positions, superposition=superposition) 

204 return round(rmsd, prec)

378def _HELP_sss_None2int(obj, cfg): 

379 """ 

380 Converts None entries of the config keys to integers: 

381 sss = [None, None, None] -> [0, max frame of MDA universe, 1] 

382 start = None -> 0 

383 stop = None -> max frame of u 

384 step = None -> 1 

385 

386 Note: 

387 this function is (probably) required when a custom function uses either 

388 a MDA universe or (distance)matrix and a misc.CONFIG class with the keys 

389 "sss", "start", "stop", "step". 

390 

391 Reason: 

392 slicing: aList[None:None:None] is equal to aList[::] 

393 get item: aList[None] does not exist 

394 

395 Args: 

396 obj (universe, matrix): MDA universe or distance(matrix) 

397 cfg (misc.CONFIG class) 

398 

399 Returns: 

400 cfg (misc.CONFIG class) 

401 """ 

402 if isinstance(obj, mda.core.universe.Universe): 

403 if "sss" in cfg.keys(): 

404 if cfg.sss[0] == None: 

405 cfg.sss[0] = 0 

406 if cfg.sss[1] == None: 

407 cfg.sss[1] = obj.trajectory.n_frames 

408 if cfg.sss[2] == None: 

409 cfg.sss[2] = 1 

410 

411 if "start" in cfg.keys() or "stop" in cfg.keys() or "step" in cfg.keys(): 

412 if cfg.start == None: 

413 cfg.start = 0 

414 if cfg.stop == None: 

415 cfg.stop = obj.trajectory.n_frames 

416 if cfg.step == None: 

417 cfg.step = 1 

418 

419 elif isinstance(obj, (np.ndarray, list)): 

420 if "sss" in cfg.keys(): 

421 if cfg.sss[0] == None: 

422 cfg.sss[0] = 0 

423 if cfg.sss[1] == None: 

424 cfg.sss[1] = len(obj) 

425 if cfg.sss[2] == None: 

426 cfg.sss[2] = 1 

427 

428 if "start" in cfg.keys() or "stop" in cfg.keys() or "step" in cfg.keys(): 

429 if cfg.start == None: 

430 cfg.start = 0 

431 if cfg.stop == None: 

432 cfg.stop = len(obj) 

433 if cfg.step == None: 

434 cfg.step = 1 

435 return cfg

440 """

509def get_shortest_RES_distances(u, sel): 

510 """ 

511 Calculates shortest RES distances (current frame) for selection of universe u. 

512 

513 .. Attention:: Displayed RES ids always start with 1 

514 

515 Args: 

516 u (universe, str): structure universe or pdb path 

517 sel (str): selection string 

518 

519 Returns: 

520 SD (arrray) 

521 | NxN Matrix with shortest RES distances 

522 | i.e.: SD[0][0] is shortest distance RES0-RES0 

523 | SD[0][1] is shortest distance RES0-RES1 

524 | SD[0][2] is shortest distance RES0-RES2 

525 | etc. 

526 SD_d (array, dtype=object) 

527 detailed List with [d_min, (RES_pair), (ATM_pair), (ATM_names)] 

528 """ 

529 if isinstance(u, str): 

530 # uc: universe copy 

531 uc = mda.Universe(u) 

532 elif isinstance(u, mda.Universe): 

533 # uc: universe copy 

534 uc = u.copy() 

535 elif isinstance(u, mda.AtomGroup): 

536 uc = u.universe.copy() 

537 _top.norm_resids(uc, verbose=True) 

538 a = _top.true_select_atoms(uc, sel) 

539 dim = len(a.residues.resids) 

540 SD = np.zeros(shape=(dim, dim)) 

541 SD_d = [] 

542 

543 for i in range(dim): 

544 for j in range(i + 1, dim): 

545 DA = _distances.distance_array(a.residues[i].atoms.positions, a.residues[j].atoms.positions) 

546 d_min = np.min(DA) 

547 d_min_index = np.unravel_index(np.argmin(DA), DA.shape) # atom indices 

548 

549 # write values to list 

550 RES_pair = (a.residues[i].resid, a.residues[j].resid) 

551 ATM_pair = (a.residues[i].atoms[d_min_index[0]].id, 

552 a.residues[j].atoms[d_min_index[1]].id) 

553 ATM_names = (a.residues[i].atoms.names[d_min_index[0]], 

554 a.residues[j].atoms.names[d_min_index[1]]) 

555 

556 SD[i][j] = d_min 

557 SD_d.append([d_min, RES_pair, ATM_pair, ATM_names]) 

558 

559 # SD contains only half of the distances (double for loop w/o double counting RES pairs ) 

560 SD_t = np.transpose(SD) # -> transpose SD 

561 SD = np.maximum(SD, SD_t) # -> take element-wise maximum 

562 

563 # convert to array with dtype=object since array contains lists, tuples and sequences 

564 SD_d = np.array(SD_d, dtype=object) 

565 return(SD, SD_d)

573def get_trendline(xdata, ydata, compress=10): 

574 """ 

575 Reduces <xdata, ydata> by <compress> factor and returns trendline data. 

576 (Trendline: sum and normalize each <compress> elements of input data) 

577 

578 Args: 

579 xdata (list, array) 

580 ydata (list, array) 

581 compress (int): compress factor (sum and normalize <compress> elements) 

582 

583 Returns: 

584 trend_xdata (list, array) 

585 reduced xdata 

586 trend_ydata (list, array) 

587 reduced ydata 

588 """ 

589 if len(xdata) != len(ydata): 

590 raise ValueError("xdata and ydata have unequal length.") 

591 

592 trend_xdata = np.add.reduceat(xdata, np.arange(0, len(xdata), compress))/float(compress) 

593 trend_ydata = np.add.reduceat(ydata, np.arange(0, len(ydata), compress))/float(compress) 

594 trend_xdata = trend_xdata.tolist() 

595 trend_ydata = trend_ydata.tolist() 

596 

597 # fix normalization of last element if data is not fully divisible by compress (ie remainder exists) 

598 remainder = len(xdata) % compress 

599 if remainder != 0:

862def _HELP_setup_bins(data, vmin=None, vmax=None, vbase=1, bins=None, n_bins=200, n_bins_autoadd=1): 

863 """ 

864 HELP function to setup bins for histogram. Returns a dictionary to update

875 n_bins, vmin, vmax, vbase

904 else: 

905 bins = cfg.bins
