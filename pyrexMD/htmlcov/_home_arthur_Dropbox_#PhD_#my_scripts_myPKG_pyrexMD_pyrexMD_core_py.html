71 def get_color_scheme(self, color, universe=None): 

72 """ 

73 Returns color scheme. 

74 

75 Args: 

76 color (str) 

77 universe (universe, atomgrp) 

78 

79 Returns: 

80 color_scheme (array) 

81 HEX colors for each RES taken from iPlayer.universe (default) or universe. 

82 

83 Accepted colors (simple): 

84 | 'red' 'black' 

85 | 'orange' 'grey_80' 

86 | 'yellow' 'grey_60' 

87 | 'green' 'grey_40' 

88 | 'cyan' 'grey_20' 

89 | 'blue' 'white' 

90 | 'purple' 

91 | 'magenta' 

92 

93 Accepted colors (complex): 

94 | 'baw': black and white 

95 | 'b/w': black and white 

96 | 'rainbow': red to magenta 

97 """ 

98 color = str(color) 

99 

100 if universe != None: 

101 length = len(universe.residues) 

102 else: 

103 length = len(self._iPlayer.universe.residues) 

104 

105 # simple color schemes 

106 color_scheme_s = {'red': ['0xFF0000'], 

107 'orange': ['0xFF9900'], 

108 'yellow': ['0xFFFF00'], 

109 'green': ['0x00FF00'], 

110 'cyan': ['0x00FFFF'], 

111 'blue': ['0x0000FF'], 

112 'purple': ['0x9900FF'], 

113 'magenta': ['0xFF00FF'], 

114 

115 'black': ['0x000000'], 

116 'grey_80': ['0x444444'], 

117 'grey_60': ['0x666666'], 

118 'grey_40': ['0x999999'], 

119 'grey_20': ['0xCCCCCC'], 

120 'white': ['0xFFFFFF'] 

121 } 

122 

123 # complex color schemes 

124 color_scheme_c = {'rainbow': [color_scheme_s['red'], 

125 color_scheme_s['orange'], 

126 color_scheme_s['yellow'], 

127 color_scheme_s['green'], 

128 color_scheme_s['cyan'], 

129 color_scheme_s['blue'], 

130 color_scheme_s['purple'], 

131 color_scheme_s['magenta']], 

132 'baw': [color_scheme_s['black'], 

133 color_scheme_s['grey_80'], 

134 color_scheme_s['grey_60'], 

135 color_scheme_s['grey_40'], 

136 color_scheme_s['grey_20'], 

137 color_scheme_s['white']], 

138 'b/w': [color_scheme_s['black'], 

139 color_scheme_s['grey_80'], 

140 color_scheme_s['grey_60'], 

141 color_scheme_s['grey_40'], 

142 color_scheme_s['grey_20'], 

143 color_scheme_s['white']] 

144 } 

145 

146 # simple 

147 if color in color_scheme_s: 

148 color_scheme = color_scheme_s[color] * length 

149 

150 # complex 

151 elif color in color_scheme_c: 

152 color_scheme = color_scheme_c[color] 

153 if length % len(color_scheme) == 0: 

154 repeat = length / len(color_scheme) 

155 else: 

156 repeat = 1 + length / len(color_scheme) 

157 color_scheme = list(np.repeat(color_scheme, repeat)) 

158 

159 # non-existing color schemes 

160 else: 

161 print("Color scheme does not exist!") 

162 

163 return(color_scheme)

165 def set_color_scheme(self, color, ci=None): 

166 """ 

167 Applies color scheme. If ci=None, then target is determinded by <iPlayer_object>.widgets.Components.description. 

168 

169 Args: 

170 color (str) 

171 universe (universe, atomgrp) 

172 

173 Returns: 

174 color_scheme (array) 

175 HEX colors for each RES taken from iPlayer.universe (default) or universe. 

176 

177 Accepted colors (simple): 

178 | 'red' 'black' 

179 | 'orange' 'grey_80' 

180 | 'yellow' 'grey_60' 

181 | 'green' 'grey_40' 

182 | cyan' 'grey_20' 

183 | 'blue' 'white' 

184 | 'purple' 

185 | 'magenta' 

186 

187 Accepted colors (complex): 

188 | 'baw': black and white 

189 | 'b/w': black and white 

190 | 'rainbow': red to magenta 

191 """ 

192 color_schemes = ['red', 'yellow', 'orange', 'green', 

193 'cyan', 'blue', 'purple', 'magenta', 

194 'black', 'grey_80', 'grey_60', 'grey_40', 

195 'grey_20', 'white', 'baw', 'b/w', 'rainbow'] 

196 color_scheme_str = None 

197 

198 # check for existing color schemes 

199 if color in color_schemes: 

200 color_scheme_str = color 

201 color = self.get_color_scheme(color) 

202 

203 # update dict / remember color scheme 

204 if ci == None: 

205 ci = self._iPlayer.widgets.Components.description.split()[-1] 

206 

207 if color_scheme_str != None: 

208 if ci == '0': 

209 self._iPlayer.widgets._dict_Component_0['Color_Scheme'] = color_scheme_str 

210 elif ci == '1': 

211 self._iPlayer.widgets._dict_Component_1['Color_Scheme'] = color_scheme_str 

212 else: 

213 if ci == '0': 

214 self._iPlayer.widgets._dict_Component_0['Color_Scheme'] = color 

215 elif ci == '1': 

216 self._iPlayer.widgets._dict_Component_1['Color_Scheme'] = color 

217 

218 # apply color scheme 

219 for ri in range(0, 5): 

220 self._iPlayer.player._set_color_by_residue(color, component_index=ci, repr_index=ri) 

221 return
